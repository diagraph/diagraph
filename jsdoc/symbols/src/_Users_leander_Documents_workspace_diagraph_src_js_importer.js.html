  2 // raphael paper
  3 var canvas;
  4 var canvas_size = { width: 100, height: 100 };
  5 
  6 // this updates the svg preview with the current text inside the 'import_text' textbox
  7 function update_preview() {
  8 	canvas.clear();
  9 	var path = canvas.path($('#import_text').val());
 10 	
 11 	// scale
 12 	var bbox = path.getBBox();
 13 	// to prevent cut off on the canvas sides, subtract one pixel for each side
 14 	var scale = Math.min((canvas_size.width-2) / bbox.width, (canvas_size.height-2) / bbox.height);
 15 	path.scale(scale, scale, 0, 0);
 16 	bbox = path.getBBox(); // new bbox after scaling
 17 	path.translate(1-bbox.x, 1-bbox.y); // same here: offset by 1px
 18 }

 20 function update_categories() {
 21 	// get categories and add them to the category combo box
 22 	$('#import_category').text('');
 23 	$.ajax({
 24 		   data: { task: 'getCategories' },
 25 		   dataType: 'json',
 26 		   success: function(json) {
 27 				for(var item in json) {
 28 					$('<option value=\'' + json[item].id + '\'>' + json[item].name + '</option>').appendTo('#import_category');
 29 				}
 30 				$('#import_category').val(0);
 31 		   }
 32 	});
 33 }

 35 $(function() {
 36 	$.ajaxSetup({type: 'post'});
 37 	
 38 	// init raphael
 39 	canvas = Raphael(document.getElementById('preview'), canvas_size.width, canvas_size.height);
 40 	
 41 	// update categories
 42 	update_categories();
 43 	
 44 	// handle svg file import
 45 	$('#import_file').change(function(event) {
 46 		var reader = new FileReader();
 47 		reader.onload = function() {
 48 			if(reader.result.length == 0) {
 49 				alert('Sorry, but you can\'t add an empty file!');
 50 				return;
 51 			}
 52 			
 53 			var svg = $($.parseXML(reader.result));
 54 			var paths = svg.find('path');
 55 			if(paths.length == 0) {
 56 				alert('Sorry, but your SVG image doesn\'t contain any paths!');
 57 				return;
 58 			}
 59 			
 60 			// clear old data
 61 			$('#import_text').text('');
 62 			$('#import_text').val('');
 63 			
 64 			// new data:
 65 			var data = '';
 66 			paths.each(function (index, path) {
 67 				var path = $(path).attr('d');
 68 				if(path) {
 69 					// replace first 'm' by 'M' if it exists and isn't preceded by an 'M'
 70 					var first_m = path.indexOf('m');
 71 					var first_M = path.indexOf('M');
 72 					if(first_m >= 0 && (first_M < 0 || first_M > first_m)) {
 73 						path = path.substring(0, first_m)+'M'+path.substring(first_m+1, path.length);
 74 					}
 75 					
 76 					data += path + '\n';
 77 				}
 78 			});
 79 			$('#import_text').val(data);
 80 			
 81 			update_preview();
 82 		}
 83 		reader.readAsBinaryString($('#import_file')[0].files[0]);
 84 	});
 85 	
 86 	// update the svg preview when the textbox changes (key input)
 87 	$('#import_text').bind('keyup', function(event) {
 88 		update_preview();
 89 	});
 90 	
 91 	// handle form submit
 92 	$('#import_form').submit(function(event) {
 93 		event.preventDefault();
 94 		
 95 		var import_name = $('#import_name').val();
 96 		var import_text = $('#import_text').val();
 97 		var import_category = $('#import_category').val();
 98 		
 99 		$.ajax({
100 		   data: { task: 'import', import_name: import_name, import_text: import_text, import_category: import_category },
101 		   dataType: 'text',
102 		   success: function(text) {
103 				if(text == 'success') {
104 					alert('Import successful!');
105 					
106 					// clear form
107 					$('#import_name').val('');
108 					$('#import_file').val('');
109 					$('#import_text').val('');
110 					$('#import_category').val(0);
111 					update_preview();
112 					
113 					return;
114 				}
115 				
116 				alert('Import failed: '+text);
117 		   }
118 		});
119 	});
120 	
121 	// handle 'add category'
122 	$('#import_add_category').click(function(event) {
123 		var category_name = prompt('Category Name:', '');
124 		if(category_name == '' || category_name.length == 0) {
125 			alert('Category name is empty!');
126 			return;
127 		}
128 		else if(category_name.length > 32) {
129 			alert('Category name is too long! (max: 32 characters)');
130 			return;
131 		}
132 		
133 		//
134 		$.ajax({
135 		   data: { task: 'addCategory', category_name: category_name },
136 		   dataType: 'text',
137 		   success: function(text) {
138 				if(text == 'success') {
139 					alert('Category \''+category_name+'\' added successfully!');
140 					update_categories();
141 					return;
142 				}
143 				
144 				alert('Failed to add category: '+text);
145 		   }
146 		});
147 	});
148 });
