  4 vonline.Sidebar = function(container) {
  5 	var that = this;
  6 	this.container = $(container).css({width: '200px'});
  7 	this.topmenu = $('<div/>').appendTo(this.container),
  8 	this.categories  = $('<div/>').appendTo(this.container).addClass('categories').css('width', '200px'),
  9 	this.extraview = $('<div/>').appendTo(this.container).addClass('extraview').hide().css('width', '200px'),
 10 	this.bottommenu  = $('<div/>').appendTo(this.container);
 11 	this.extraviewVisible = false;
 12 	
 13 	function onResize() {
 14 		var height = $(window).height() - that.topmenu.height() - that.bottommenu.height();
 15 		that.categories.height(height);
 16 		that.extraview.height(height).css({position: 'absolute', width: '200px', left: '200px', top: that.topmenu.height()+'px'});
 17 	}
 18 	$(window).bind('resize', onResize);
 19 }
 20 
 21 vonline.Sidebar.prototype.setTopMenu = function(menu) {
 22 	this.topmenu.html(menu.getHTML());
 23 	this._topmenu = menu;
 24 }
 25 
 26 vonline.Sidebar.prototype.getTopMenu = function() {
 27 	return this._topmenu;
 28 }
 29 
 30 vonline.Sidebar.prototype.setBottomMenu = function(menu) {
 31 	this.bottommenu.html(menu.getHTML());
 32 }
 33 
 34 vonline.Sidebar.prototype.addCategory = function(category) {
 35 	this.categories.append(category.getHTML());
 36 }

 38 /**
 39  * @param {boolean} mode
 40  * @return extraview reference to dom-object
 41  */
 42 vonline.Sidebar.prototype.setExtraView = function(mode) {
 43 	this.extraviewVisible = mode;
 44 	if (mode) {
 45 		this.container.css('width', '400px');
 46 		return this.extraview.show();
 47 	}
 48 	else {
 49 		this.container.css('width', '200px');
 50 		this.extraview.hide();
 51 	}
 52 }
 53 
 54 vonline.Sidebar.prototype.isExtraViewVisible = function() {
 55 	return this.extraviewVisible;
 56 }
