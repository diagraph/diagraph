  5 /**
  6  * parses url query string to javascript array
  7  */
  8 jQuery.parseQueryString = function() {
  9 	var query = window.location.search.substring(1),
 10 	vars = query.split("&"),
 11 	query = {};
 12 	for (var i = 0, len = vars.length; i < len; i++) {
 13 		vars[i] = vars[i].split('=');
 14 		query[vars[i][0]] = vars[i][1];
 15 	}
 16 	return query;
 17 }

 19 /**
 20  * returns an array without the value elements
 21  * @param {array} array
 22  * @param {mixed} value
 23  */
 24 jQuery.without = function(array, value) {
 25 	return jQuery.grep(array, function(val) {
 26 		return val != value;
 27 	});
 28 }

 30 /**
 31  * formats Date to SQL time, e.g. 2011-02-04 10:52:47
 32  * @param {Date} date
 33  * @returns {String}
 34  */
 35 function format_time(date) {
 36 	return date.getFullYear() + '-'
 37 	 + format_decimal(date.getMonth()+1, 2) + '-'
 38 	 + format_decimal(date.getDate(), 2) + ' ' 
 39 	 + format_decimal(date.getHours(), 2) + ':' 
 40 	 + format_decimal(date.getMinutes(), 2) + ':'
 41 	 + format_decimal(date.getSeconds(), 2);
 42 }

 44 /**
 45  * adds leading zeros to a number
 46  * @param number
 47  * @param decimals
 48  * @returns
 49  */
 50 function format_decimal(number, decimals) {
 51 	var str = String(number);
 52 	if (str.length < decimals) {
 53 		return format_decimal('0'+str, decimals);
 54 	}
 55 	return str;
 56 }
