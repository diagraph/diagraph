1 /**
  2  * @namespace
  3  * @augments vonline.Base
  4  */
  5 vonline.Rectangle = function(data) {
  6 	// inherit default values
  7 	this.data = $.extend({}, this.data, data);
  8 }
  9 
 10 vonline.Rectangle.prototype = new vonline.Base();
 11 
 12 /**
 13  * Creates the object on the given canvas
 14  * @param {vonline.Canvas} canvas
 15  */
 16 vonline.Rectangle.prototype.createObject = function(canvas) {
 17 	return canvas.getPaper().rect(this.data.x, this.data.y, this.data.width, this.data.height);
 18 }
 19 
 20 vonline.Rectangle.prototype.setScale = function(x, y, origX, origY) {
 21 	this.obj.scale(x, y, origX, origY);
 22 	this.obj.resetScale();  // get cumulative scaling
 23 	this.data.x = this.obj.attr('x');
 24 	this.data.y = this.obj.attr('y');
 25 	this.data.width = this.obj.attr('width');
 26 	this.data.height = this.obj.attr('height');
 27 	$(this.obj.node).trigger('changed');
 28 }
