1 /**
  2  * @namespace
  3  */
  4 vonline.Category = function(name, id) {
  5 	var that = this;
  6 	this.id = id;
  7 	this.container = $('<div/>').addClass('category');
  8 	this.header = $('<div/>').addClass('header').appendTo(this.container).text(name);
  9 	this.handle = $('<span/>').addClass('handle active').appendTo(this.header).click(function() {
 10 		that.toggle();
 11 	});
 12 	this.button = $('<span/>').addClass('button').appendTo(this.header);
 13 	this.body = $('<div/>').appendTo(this.container);
 14 }
 15 
 16 vonline.Category.prototype.getHTML = function() {
 17 	return this.container;
 18 }
 19 
 20 /**
 21  * @param {vonline.CategoryItem}
 22  */
 23 vonline.Category.prototype.add = function(element) {
 24 	this.body.append(element.getHTML());
 25 }
 26 
 27 vonline.Category.prototype.open = function() {
 28 	this.body.show();
 29 	this.handle.addClass('active');
 30 }
 31 
 32 vonline.Category.prototype.close = function() {
 33 	this.body.hide();
 34 	this.handle.removeClass('active');
 35 }
 36 
 37 vonline.Category.prototype.toggle = function() {
 38 	if (this.handle.hasClass('active')) {
 39 		this.close();
 40 	}
 41 	else {
 42 		this.open();
 43 	}
 44 }
 45 
 46 vonline.Category.prototype.detach = function() {
 47 	this.container.detach();
 48 }
 49 
 50 /**
 51  * @param {string} mode
 52  * @param {function} onclick
 53  */
 54 vonline.Category.prototype.setMode = function(mode, onclick) {
 55 	this.button.removeClass('add').removeClass('remove');
 56 	if (!mode) {
 57 		this.button.unbind('click');
 58 	}
 59 	else {
 60 		this.button.addClass(mode);
 61 		this.button.click(function() {
 62 			onclick();
 63 		});
 64 	}
 65 }
