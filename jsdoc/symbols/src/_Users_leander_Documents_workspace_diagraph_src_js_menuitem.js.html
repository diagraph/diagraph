1 /**
  2  * menu item, containing an image and a click-event handler
  3  * @namespace
  4  */
  5 vonline.MenuItem = function(name, src, clickHandler) {
  6 	var that = this,
  7 	li = $('<li/>').attr('title', name).addClass('item'),
  8 	image = $('<img/>').appendTo(li).bind('load', function() {
  9 		$(window).trigger('resize'); // layout changes if images are loaded (see sidebar)
 10 	}),
 11 	enabled = false;
 12 	
 13 	function onclick() {
 14 		// get the correct context (this object instead of the li-element)
 15 		clickHandler.apply(that);
 16 	}
 17 	
 18 	this.getHTML = function() {
 19 		this.enable();
 20 		return li;
 21 	}
 22 	
 23 	this.disable = function() {
 24 		if (enabled) {
 25 			image.attr('src', src + '_disabled.png');
 26 			li.removeClass('enabled');
 27 			li.unbind('click', onclick);
 28 			enabled = false;
 29 		}
 30 	}
 31 	
 32 	this.enable = function() {
 33 		if (!enabled) {
 34 			image.attr('src', src + '.png');
 35 			li.addClass('enabled');
 36 			li.click(onclick);
 37 			enabled = true;
 38 		}
 39 	}
 40 }
