  4 vonline.Dialog = function(options) {
  5 	this.options = this.getOptions(options);
  6 	this.background = $('<div/>').addClass('dialog_background').appendTo('body');
  7 	this.container = $('<div/>').addClass('dialog_container').appendTo(this.background);
  8 	this.content = $('<div/>').html(this.options.text).appendTo(this.container);
  9 	this.buttons = $('<div/>').addClass('dialog_buttons').appendTo(this.container);
 10 	
 11 	this.createButtons();
 12 	
 13 	this.options.init();
 14 	
 15 	var that = this;
 16 	function onResize() {
 17 		that.container.css({
 18 			top: $('body').height() / 2 - that.container.outerHeight() + 'px',
 19 			left: ($('body').width() - that.container.outerWidth()) / 2 + 'px'
 20 		});
 21 	}
 22 	onResize();
 23 }
 24 
 25 vonline.Dialog.prototype.defaultOptions = {text: '', init: function() {}}
 26 
 27 vonline.Dialog.prototype.getOptions = function(options) {
 28 	return $.extend({}, this.defaultOptions, options);
 29 }
 30 
 31 vonline.Dialog.prototype.createButtons = function() {
 32 	var that = this;
 33 	$('<button/>').addClass('dialog_cancel cancel').text('close').appendTo(this.buttons).click(function() {
 34 		that.close();
 35 	});
 36 }
 37 
 38 vonline.Dialog.prototype.close = function() {
 39 	this.background.detach();
 40 	this.container.detach();
 41 }

 43 /**
 44  * @namespace
 45  */
 46 vonline.ConfirmDialog = function(options) {
 47 	vonline.Dialog.apply(this, [options]);
 48 };
 49 vonline.ConfirmDialog.prototype = new vonline.Dialog();
 50 
 51 vonline.ConfirmDialog.prototype.defaultOptions = $.extend({}, vonline.Dialog.prototype.defaultOptions, {confirm: function() {}, cancel: function() {}});
 52 
 53 vonline.ConfirmDialog.prototype.createButtons = function() {
 54 	var that = this;
 55 	this.cancel = $('<button/>').addClass('dialog_cancel cancel').text('cancel').appendTo(this.buttons).click(function() {
 56 		that.close();
 57 		that.options.cancel();
 58 	})
 59 	this.submit = $('<button/>').addClass('dialog_ok ok').text('ok').appendTo(this.buttons).click(function() {
 60 		that.close();
 61 		that.options.confirm();
 62 	});
 63 
 64 	function keyHandler (event) {
 65 		event.stopPropagation();
 66 		if (event.keyCode == 13) {
 67 			that.submit.triggerHandler('click');
 68 			$(window).unbind('keydown', keyHandler);
 69 			$(window).unbind('keypress', keyHandler);
 70 		}
 71 		else if (event.keyCode == 27) {
 72 			that.cancel.triggerHandler('click');
 73 			$(window).unbind('keydown', keyHandler);
 74 			$(window).unbind('keypress', keyHandler);
 75 		}
 76 	}
 77 	$(window).bind('keydown', keyHandler);
 78 	$(window).bind('keypress', keyHandler);
 79 }

 81 /**
 82  * @namespace
 83  */
 84 vonline.InputDialog = function(options) {
 85 	var that = this;
 86 	
 87 	options = this.getOptions(options);
 88 	var input = $('<input/>').attr({type: 'text', value: options.value});
 89 	options.text = $('<div/>').append($('<div/>').text(options.text)).append($('<div/>').html(input));
 90 	options.init = function() {
 91 		input.focus();
 92 		that.submit.unbind('click');
 93 		that.submit.click(function() {
 94 			that.close();
 95 			that.options.confirm(input.val());
 96 		});
 97 	}
 98 	vonline.ConfirmDialog.apply(this, [options]);
 99 };
100 
101 vonline.InputDialog.prototype = new vonline.ConfirmDialog();
102 
103 vonline.InputDialog.prototype.defaultOptions = $.extend({}, vonline.ConfirmDialog.prototype.defaultOptions, {value: ''});
