1 /**
  2  * Class for managing the svg elements
  3  * @namespace

  5 vonline.Canvas = function() {
  6 	var that = this;
  7 	this.container = $('#canvas');
  8 	this.paper = Raphael('canvas', this.container.width(), this.container.height()).initZoom();
  9 	this.objects = [];
 10 	this.maxId = 0;
 11 	this.selection = new vonline.Selection(this);
 12 	this.offset = {x:0, y:0};
 13 	this.viewport = {width:0, height:0}
 14 	this.zoom = 1.0;
 15 	this.initRectangleSelection();
 16 	this.initDragging();
 17 	
 18 	function onResize() {
 19 		var sidebarwidth = $('#sidebar').width(),
 20 		documentwidth = ($(window).width() - that.offset.x) * 1.5,
 21 		documentheight = ($(window).height() - that.offset.y) * 1.5;
 22 		that.viewport.width = $(window).width() - sidebarwidth;
 23 		that.viewport.height = $(window).height();
 24 		that.container.css({width: (documentwidth-sidebarwidth)+'px', height: documentheight + 'px', marginLeft: sidebarwidth+'px'});
 25 		that.paper.setSize(that.container.width(), that.container.height());
 26 	}
 27 	$(window).bind('resize', onResize);
 28 }

 34 /**
 35  * loads a document from its json representation
 36  * @param {array} json array containing the components
 37  * @exmaple
 38  * vonline.document.canvas.load([{type:'rectangle', id:1, scaleX:1, scaleY:1, x: 100, y:50}]);
 39  */
 40 vonline.Canvas.prototype.load = function(json) {
 41 	
 42 	this.clear();
 43 	
 44 	for (var i = 0, count = json.length; i < count; i++) {
 45 		var object = this.createObject(json[i]);
 46 		if (object.data.id > this.maxId) {
 47 			this.maxId = object.data.id;
 48 		}
 49 		this.add(object);
 50 	}
 51 }

 53 vonline.Canvas.prototype.createObject = function(data) {
 54 	switch (data.type) {
 55 	case 'rectangle':
 56 		return new vonline.Rectangle(data);
 57 	break;
 58 	case 'connection':
 59 		return new vonline.Connection(data);
 60 	break;
 61 	case 'annotation':
 62 		return new vonline.Annotation(data);
 63 	break;
 64 	case 'path':
 65 		return new vonline.Path(data);
 66 	break;
 67 	}
 68 }

 70 /**
 71  * @param {vonline.Base} obj
 72  */
 73 vonline.Canvas.prototype.add = function(obj) {
 74 	if (!obj.data.id) {
 75 		obj.data.id = ++this.maxId;
 76 	}
 77 	obj.setCanvas(this);
 78 	this.objects.push(obj);
 79 }

 81 /**
 82  * @param {vonline.Base} obj
 83  */
 84 vonline.Canvas.prototype.remove = function(obj) {
 85 	obj.setCanvas(null);
 86 	this.objects = $.grep(this.objects, function(object) {
 87 		return object != obj;
 88 	});
 89 }

 91 /**
 92  * Deletes all object from canvas
 93  */
 94 vonline.Canvas.prototype.clear = function() {
 95 	this.selection.clear();
 96 	for (var i = this.objects.length - 1; i >= 0; i--) {
 97 		this.remove(this.objects[i]);
 98 	}
 99 }

126 vonline.Canvas.prototype.initRectangleSelection = function() {
127 	var that = this;
128 	this.container.mousedown(function(event) {
129 		event.preventDefault();
130 		if (event.which == 3) {
131 			return;
132 		}
133 		event = that.normalizeEvent(event);
134 		var x = event.offsetX,
135 			y = event.offsetY,
136 			rect = that.paper.rect(x, y).attr({stroke: 'blue', fill: 'rgba(0,100,255,.5)'});
137 		
138 		var moveEvent = function(event) {
139 			event = that.normalizeEvent(event);
140 			var width = (event.offsetX - x),
141 				height = (event.offsetY - y);
142 			if (width < 0) {
143 				rect.attr('width', -width);
144 				rect.attr('x', event.offsetX);
145 			}
146 			else {
147 				rect.attr('width', width);
148 				rect.attr('x', x);
149 			}
150 			if (height < 0) {
151 				rect.attr('height', -height);
152 				rect.attr('y', event.offsetY);
153 			}
154 			else {
155 				rect.attr('height', height);
156 				rect.attr('y', y);
157 			}
158 		}
159 		that.container.mousemove(moveEvent);
160 		
161 		that.container.one('mouseup', function(event) {
162 			that.container.unbind('mousemove', moveEvent);
163 			
164 			if (!event.shiftKey) {
165 				that.selection.clear();
166 			}
167 			
168 			var bbox = rect.getBBox();
169 			rect.remove();
170 			for (var i = 0, count = that.objects.length; i < count; i++) {
171 				if (that.objects[i].fitsIn(bbox)) {
172 					that.selection.add(that.objects[i]);
173 				}
174 			}
175 		})
176 	});
177 }

179 /**
180  * normalizes the event.offsetX and event.offsetY value
181  */
182 vonline.Canvas.prototype.normalizeEvent = function(event) {
183 	var offset = this.container.offset();
184 	if (!event.offsetX) {
185 		event.offsetX = event.pageX - offset.left;
186 	}
187 	if (!event.offsetY) {
188 		event.offsetY = event.pageY - offset.top;
189 	}
190 	return event;
191 }

193 vonline.Canvas.prototype.initDragging = function() {
194 	var that = this;
195 	this.container.bind('contextmenu', function(event) {
196 		event.preventDefault();
197 	});
198 	this.container.mousedown(function(event) {
199 		event.preventDefault();
200 		if (event.which != 3) {
201 			return;
202 		}
203 		var x = event.pageX,
204 		y = event.pageY,
205 		deltaX = 0,
206 		deltaY = 0;
207 		
208 		var moveEvent = function(event) {
209 			that.container.css('cursor', 'move');
210 			deltaX = x - event.pageX;
211 			deltaY = y - event.pageY;
212 			that.container.css({
213 				left: (that.offset.x - deltaX) >= 0 ? 0 : (that.offset.x - deltaX) + 'px',
214 				top: (that.offset.y - deltaY) >= 0 ? 0 : (that.offset.y - deltaY) + 'px'
215 			});
216 		}
217 		that.container.mousemove(moveEvent);
218 		
219 		that.container.one('mouseup', function(event) {
220 			that.container.unbind('mousemove', moveEvent);
221 			that.container.css('cursor', 'auto');
222 			
223 			that.setOffset(that.offset.x - deltaX, that.offset.y - deltaY);
224 			
225 			$(window).trigger('resize');
226 		})
227 	});
228 }

230 vonline.Canvas.prototype.setOffset = function(x, y) {
231 	x = x >= 0 ? 0 : x;
232 	y = y >= 0 ? 0 : y
233 	this.container.css({
234 		left: x + 'px',
235 		top: y + 'px'
236 	});
237 	this.offset.x = x;
238 	this.offset.y = y;

241 vonline.Canvas.prototype.setZoom = function(zoom) {
242 	if (zoom == 'fit') {
243 		this.paper.setZoom(1);
244 		var objects = this.objects;
245 		var tofit = [];
246 		$.each(objects, function(i, obj) {
247 			tofit.push(obj.obj);
248 		});
249 		tofit = this.paper.set(tofit);
250 		tofit = tofit.getBBox();
251 		zoom = Math.min(this.viewport.width / (tofit.width + 25), this.viewport.height / (tofit.height + 25));
252 		this.paper.setZoom(zoom);
253 		this.setOffset(-(tofit.x-10) * zoom, -(tofit.y-10) * zoom);
254 		$(window).trigger('resize');
255 		return;
256 	}
257 	this.paper.setZoom(zoom);
258 	
259 	var delta = zoom / this.zoom;
260 	if (delta > 1) {
261 		this.setOffset(
262 			this.offset.x * delta - this.viewport.width / delta,
263 			this.offset.y * delta - this.viewport.height / delta
264 		);
265 	}
266 	else if (delta < 1) {
267 		this.setOffset(
268 			(this.offset.x + this.viewport.width * delta) * delta,
269 			(this.offset.y + this.viewport.height * delta) * delta
270 		);
271 	}
272 	
273 	this.zoom = zoom;
274 	$(window).trigger('resize');
275 	$(vonline.events).trigger('zoom');
276 }
