1 /**
  2  * @namespace
  3  * @augments vonline.Base
  4  */
  5 vonline.Connection = function(data) {
  6 	// inherit default values
  7 	this.data = $.extend({}, this.data, data);
  8 	this.data.color = 'none';
  9 	this.resizeable = false;
 10 }
 11 
 12 vonline.Connection.prototype = new vonline.Base();
 13 
 14 vonline.Connection.prototype.createObjectId = function() {
 15 	return 'connection_'+this.data.id;
 16 }
 17 
 18 /**
 19  * Creates the object on the given canvas
 20  * @param {vonline.Canvas} canvas
 21  */
 22 vonline.Connection.prototype.createObject = function(canvas) {
 23 	var that = this;
 24 	
 25 	this.sourceObject = canvas.getObjectById(this.data.connect[0]);
 26 	this.targetObject = canvas.getObjectById(this.data.connect[1]);
 27 	
 28 	// event handler
 29 	$(this.sourceObject.obj.node).bind('changed', function() {
 30 		that.updatePath();
 31 	});
 32 	$(this.targetObject.obj.node).bind('changed', function() {
 33 		that.updatePath();
 34 	});
 35 	
 36 	return canvas.getPaper().path(this.getPath()).toBack().attr('stroke-width', 2);
 37 }
 38 
 39 vonline.Connection.prototype.getPath = function() {
 40 	var sourceBBox = this.sourceObject.obj.getBBox(),
 41 	targetBBox = this.targetObject.obj.getBBox();
 42 	return vonline.Connection.computePath(sourceBBox.x+sourceBBox.width/2, sourceBBox.y+sourceBBox.height/2,
 43 										  targetBBox.x+targetBBox.width/2, targetBBox.y+targetBBox.height/2);
 44 }
 45 
 46 vonline.Connection.prototype.updatePath = function() {
 47 	this.obj.attr('path', this.getPath());
 48 }
 49 
 50 /**
 51  * Static method to compute/generate a path between the two points (x1, y1) and (x2, y2)
 52  * @param {float} x1
 53  * @param {float} y1
 54  * @param {float} x2
 55  * @param {float} y2
 56  */
 57 vonline.Connection.computePath = function(x1, y1, x2, y2) {
 58 	var path = 'M0,0';
 59 	var start = { x: x1, y: y1 };
 60 	var end = { x: x2, y: y2 };
 61 	if(Math.abs(start.x - end.x) < 50 || Math.abs(start.y - end.y) < 50) {
 62 		// if angle is too shallow, use a straight line
 63 		path = 'M' + start.x + ' ' + start.y +
 64 			   'L' + end.x + ' ' + end.y;
 65 	}
 66 	else {
 67 		// else: use a nice bezier curve
 68 		var c = { x: (end.x - start.x)/4, y: (end.y - start.y)/4 };
 69 		path = 'M' + start.x + ' ' + start.y +
 70 		       'C' + (start.x+c.x*3) + ' ' + (start.y+c.y) + ' ' + (start.x+c.x) + ' ' + (start.y+c.y*3) + ' ' + end.x + ' ' + end.y;
 71 	}
 72 	
 73 	return path;
 74 }
