1 /**
  2  * application class, cares on initialization and maintenance
  3  * @namespace
  4  * @param {number} id the document id
  5  */
  6 vonline.Document = function(id) {
  7 	this.id = id;
  8 	// storage for undo-/redoable actions
  9 	this.undoList = [];
 10 	this.redoList = [];
 11 	// observe if command was executed
 12 	var that = this;
 13 	vonline.events.bind('commandexec', function(event, command) {
 14 		that.undoList.push(command);
 15 		that.redoList = [];
 16 		that.updateMenu();
 17 		that.saveItem.enable();
 18 	});
 19 	
 20 	this.transport = new vonline.Transport(this);
 21 	
 22 	// initialization of UI
 23 	this.sidebar = new vonline.Sidebar('#sidebar');
 24 	
 25 	// init top menu
 26 	this.sidebar.setTopMenu(this.initTopMenu());
 27 	this.updateMenu();
 28 	this.snapshotHistory = new vonline.SnapshotHistory(this, this.sidebar, this.transport);
 29 
 30 	// init bottom menu
 31 	this.sidebar.setBottomMenu(this.initBottomMenu());
 32 	
 33 	this.loadCategories();
 34 	
 35 	this.canvas = new vonline.Canvas();
 36 	
 37 	vonline.events.bind('drop', function(event, data) {
 38 		var command = new vonline.CreateCommand(that.canvas, data);
 39 		command.execute();
 40 		vonline.events.trigger('commandexec', command);
 41 	});
 42 	
 43 	// load latest snapshot
 44 	this.loadSnapshot(-1);
 45 }

 47 /**
 48  * initializes the top-menu
 49  * @return {vonline.Menu}
 50  */
 51 vonline.Document.prototype.initTopMenu = function() {
 52 	var that = this,  // reference to current context (need to get the correct scope in click-hanlder functions)
 53 		topmenu = new vonline.Menu();
 54 	topmenu.addItem(new vonline.MenuItem('open an other document', 'images/menu/open_document_view', function() {
 55 		that.openDocumentView();
 56 	}));
 57 	this.undoItem = new vonline.MenuItem('undo last action', 'images/menu/undo', function() {
 58 		that.undoCommand();
 59 	});
 60 	topmenu.addItem(this.undoItem);
 61 	this.redoItem = new vonline.MenuItem('redo previous action', 'images/menu/redo', function() {
 62 		that.redoCommand();
 63 	});
 64 	topmenu.addItem(this.redoItem);
 65 	this.saveItem = new vonline.MenuItem('save a snapshot of the current document', 'images/menu/save', function() {
 66 		that.saveSnapshot();
 67 	});
 68 	topmenu.addItem(this.saveItem);
 69 	topmenu.addItem(new vonline.MenuItem('view the history of the current document', 'images/menu/open_history', function() {
 70 		that.snapshotHistory.toggle();
 71 	}));
 72 	return topmenu;
 73 }

 75 /**
 76  * initializes the bottom-menu
 77  * @return {vonline.Menu}
 78  */
 79 vonline.Document.prototype.initBottomMenu = function() {
 80 	var bottommenu = new vonline.Menu(),
 81 	that = this;
 82 	
 83 	bottommenu.addItem(new vonline.MenuItem('edit categories', 'images/menu/open_category_edit_view', function() {
 84 		if (that.categoryEditView.toggle()) {
 85 			that.transport.saveCategories(that.categoryEditView.getCategories());
 86 		}
 87 	}));
 88 	bottommenu.addItem(new vonline.MenuItem('zoom in', 'images/menu/zoom_in', function() {
 89 		var zoom = that.canvas.getZoom();
 90 		if (zoom * 2 <= 16) {
 91 			that.canvas.setZoom(zoom * 2);
 92 		}
 93 	}));
 94 	bottommenu.addItem(new vonline.MenuItem('zoom out', 'images/menu/zoom_out', function() {
 95 		var zoom = that.canvas.getZoom();
 96 		if (zoom / 2 >= 0.125) {
 97 			that.canvas.setZoom(zoom / 2);
 98 		}
 99 	}));
100 	bottommenu.addItem(new vonline.MenuItem('zoom to original', 'images/menu/zoom_original', function() {
101 		that.canvas.setZoom(1);
102 		that.canvas.setOffset(0, 0);
103 	}));
104 	bottommenu.addItem(new vonline.MenuItem('zoom fit best', 'images/menu/zoom_fit_best', function() {
105 		that.canvas.setZoom('fit');
106 	}));
107 	return bottommenu;
108 }

110 /**
111  * disable undo/redo-button if there is nothing to undo/redo
112  */
113 vonline.Document.prototype.updateMenu = function() {
114 	if (this.undoList.length == 0) {
115 		this.undoItem.disable();
116 	}
117 	else {
118 		this.undoItem.enable();
119 	}
120 	if (this.redoList.length == 0) {
121 		this.redoItem.disable();
122 	}
123 	else {
124 		this.redoItem.enable();
125 	}
126 	vonline.events.trigger('canvaschanged');
127 }

129 /**
130  * redirect browser to document list
131  */
132 vonline.Document.prototype.openDocumentView = function() {
133 	window.location.href = 'index.php';
134 }

136 /**
137  * undo action (if possible)
138  */
139 vonline.Document.prototype.undoCommand = function() {
140 	if (this.undoList.length > 0) {
141 		var command = this.undoList.pop();
142 		command.undo();
143 		this.redoList.push(command);
144 		this.updateMenu();
145 	}
146 }
147 
148 /**
149  * redo action (if possible)
150  */
151 vonline.Document.prototype.redoCommand = function() {
152 	if (this.redoList.length > 0) {
153 		var command = this.redoList.pop();
154 		command.execute();
155 		this.undoList.push(command);
156 		this.updateMenu();
157 	}
158 }

160 /**
161  * loads specific snapshot and display the corresponding document
162  * @param {number} id the snapshot id (-1 for latest snapshot)
163  */
164 vonline.Document.prototype.loadSnapshot = function(id) {
165 	var that = this;
166 	this.transport.loadSnapshot(id, function(json) {
167 		// load objects
168 		that.canvas.load(json.objects);
169 		that.undoList = [];
170 		that.redoList = [];
171 		that.updateMenu();
172 	});
173 }

175 /**
176  * save a snapshot of the current document / send document data to server
177  */
178 vonline.Document.prototype.saveSnapshot = function() {
179 	var that = this;
180 	
181 	documentData = JSON.stringify({objects: this.canvas.exportJSON()});
182 	// TODO: add other stuff that needs to be saved
183 	
184 	this.transport.saveSnapshot(documentData, function(data) {
185 		var status = 'Snapshot saved ' + format_time(new Date());
186 		vonline.notification.add(status);
187 		vonline.events.trigger('snapshotsaved');
188 	});
189 	
190 	this.saveItem.disable();
191 }

193 /**
194  * load the categories for the document
195  */
196 vonline.Document.prototype.loadCategories = function() {
197 	var that = this;
198 	this.transport.loadCategories(function(json) {
199 		var visible = [],
200 		notvisible = [];
201 		
202 		for (var name in json) {
203 			var category = new vonline.Category(name, json[name].id)
204 			for (var item in json[name].elements) {
205 				category.add(new vonline.CategoryItem(item, json[name].elements[item], that.canvas));
206 			}
207 			
208 			if (json[name].show) {
209 				visible.push(category);
210 				that.sidebar.addCategory(category);
211 			}
212 			else {
213 				notvisible.push(category);
214 			}
215 		}
216 		
217 		that.categoryEditView = new vonline.CategoryEditView(that.sidebar, visible, notvisible);
218 	});
219 }
