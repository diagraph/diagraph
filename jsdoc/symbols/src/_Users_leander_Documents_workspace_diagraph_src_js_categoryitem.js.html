1 /**
  2  * @namespace
  3  */
  4 vonline.CategoryItem = function(name, data, canvas) {
  5 	var that = this;
  6 	this.container = $('<div/>').addClass('item').attr('title', name);
  7 	this.size = 50;
  8 	this.padding = 20;
  9 	this.data = null;
 10 	
 11 	this.canvas = Raphael(this.container[0], this.size + 2*this.padding, this.size + 2*this.padding);
 12 	switch (data) {
 13 	case 'rectangle':
 14 		this.canvas.rect(this.padding, this.padding, this.size, this.size);
 15 		this.data = {type: 'rectangle'};
 16 		break;
 17 	default: // path
 18 		var path = this.canvas.path(data),
 19 		bbox = path.getBBox(),
 20 		// scale
 21 		scale = Math.min(this.size / bbox.width, this.size / bbox.height);
 22 		path.scale(scale, scale, 0, 0);
 23 		bbox = path.getBBox(); // new bbox after scaling
 24 		path.translate(this.padding-bbox.x, this.padding-bbox.y);
 25 		this.data = {type: 'path', path: path.attr('path')}
 26 		break;
 27 	}
 28 	
 29 	this.container.mousedown(function(event) {
 30 		var x = event.pageX,
 31 			y = event.pageY,
 32 			offset = that.container.offset(),
 33 			element = that.container.clone().css({position:'absolute', zIndex:2, top: offset.top+'px', left: offset.left+'px'}).appendTo(document.body);
 34 		that.wasDragging = false;
 35 		var dragEvent = function(event) {
 36 			that.wasDragging = true;
 37 			element.css({left: (offset.left + event.pageX - x)+'px', top: (offset.top + event.pageY - y)+'px'});
 38 		}
 39 		$(window).mousemove(dragEvent);
 40 		$(window).one('mouseup', function(event) {
 41 			if (that.wasDragging) {
 42 				that.data.x = offset.left + event.pageX - x - $('#sidebar').width() + that.padding - canvas.offset.x;
 43 				that.data.y = offset.top + event.pageY - y + that.padding - canvas.offset.y;
 44 				if (that.data.x > 0) {
 45 					// see vonline.Document
 46 					vonline.events.trigger('drop', that.data);
 47 				}
 48 			}
 49 			else {
 50 				that.data.x = $('#canvas').width() / 2 - that.size / 2;
 51 				that.data.y = $('#canvas').height() / 2 - that.size / 2
 52 				// see vonline.Document
 53 				vonline.events.trigger('drop', that.data);
 54 			}
 55 			that.wasDragging = false;
 56 			$(window).unbind('mousemove', dragEvent);
 57 			element.detach();
 58 		});
 59 	});
 60 }
 61 
 62 vonline.CategoryItem.prototype.getHTML = function() {
 63 	return this.container;
 64 }
