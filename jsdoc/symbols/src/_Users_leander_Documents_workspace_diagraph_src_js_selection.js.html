  4 vonline.Selection = function(canvas) {
  5 	this.canvas = canvas;
  6 	this.data = [];
  7 	this.resize = [];
  8 	this.obj = canvas.getPaper().set();
  9 	this.padding = 20;
 10 	this.resizeBox = null;
 11 	this.handlePath = 'M-20,0 L-10,-10 L-10,-5 L10,-5 L10,-10 L20,0 L10,10 L10,5 L-10,5 L-10,10z';
 12 	
 13 	var that = this;
 14 	vonline.events.bind('canvaschanged', function() {
 15 		that.updateResizeBox();
 16 	});
 17 	vonline.events.bind('zoom', function() {
 18 		that.updateResizeBox();
 19 	});
 20 	
 21 	function deleteObjFunc(event) {
 22 		if ($('input:focus').length) {
 23 			return;
 24 		}
 25 		// delete or backspace key pressed
 26 		if (event.keyCode == 46 || event.keyCode == 8) {
 27 			// also: prevent browser from navigating backwards
 28 			event.preventDefault();
 29 			event.stopPropagation();
 30 			
 31 			var command = new vonline.DeleteCommand(canvas, that.data);
 32 			command.execute();
 33 			vonline.events.trigger('commandexec', command);
 34 			
 35 			that.clear();
 36 		}
 37 	};
 38 	// bind to keypress and keydown for compatibility reasons
 39 	$(window).bind('keypress', deleteObjFunc);
 40 	$(window).bind('keydown', deleteObjFunc);
 41 }

 43 /**
 44  * @param {vonline.Base} object
 45  */
 46 vonline.Selection.prototype.add = function(object) {
 47 	var that = this;
 48 	this.data.push(object);
 49 	if (object.isResizeable()) {
 50 		this.obj.push(object.obj);
 51 		this.resize.push(object);
 52 	}
 53 	
 54 	object.setSelection(true, this.resize.length);
 55 	object.setDragEventMode(this.resize, function() {
 56 		that.updateResizeBox();
 57 	});
 58 	
 59 	this.updateResizeBox();
 60 	
 61 	vonline.events.trigger('selection_changed', this.resize.length);
 62 }

 64 /**
 65  * @param {vonline.Base} object
 66  */
 67 vonline.Selection.prototype.remove = function(object) {
 68 	this.data = $.without(this.data, object);
 69 	this.resize = $.without(this.resize, object);
 70 	var hold = $.without(this.obj, object.obj);
 71 	this.obj = this.canvas.getPaper().set(hold);
 72 	
 73 	vonline.events.trigger('selection_changed', this.resize.length);
 74 	
 75 	object.setSelection(false);
 76 	object.setDragEventMode(object);
 77 	
 78 	for (var i = 0, len = this.data.length; i < len; i++) {
 79 		this.data[i].setDragEventMode(this.resize);
 80 	}
 81 
 82 	this.updateResizeBox();
 83 }

 89 	if ($.inArray(object, this.data) != -1) {

103 /**
104  * creates the resize-box and handles resizing
105  * <br/><strong>Needs refactoring!</strong>
106  */
107 vonline.Selection.prototype.updateResizeBox = function() {
108 	var bbox = this.obj.getBBox(),
109 	that = this;
110 	
111 	if (this.resizeBox) {
112 		this.resizeBox.remove();
113 		for (var e in this.handles) {
114 			this.handles[e].remove();
115 		}
116 		this.scaleInfo.remove();
117 	}
118 	if (this.resize.length > 0) {
119 		this.resizeBox = this.canvas.getPaper().rect(
120 			bbox.x - this.padding / 2,
121 			bbox.y - this.padding / 2,
122 			bbox.width + this.padding,
123 			bbox.height + this.padding
124 		).attr('stroke', 'blue').attr('stroke-width', 1);
125 		var handle = this.canvas.getPaper().path(this.handlePath).attr({fill: 'blue', stroke: 'none'}).hide();
126 		var handlebox = handle.getBBox();
127 		var angle = -this.data[0].data.rotation * Math.PI/180; 
128 		
129 		//
130 		var handlePositions = {
131 			nw: { x: bbox.x - this.padding + handlebox.width/3, y: bbox.y - this.padding + handlebox.height/1.5 },
132 			n: { x: bbox.x + bbox.width/2, y: bbox.y - this.padding + handlebox.height/2 },
133 			ne: { x: bbox.x + bbox.width + this.padding - handlebox.width/3, y: bbox.y - this.padding + handlebox.height/1.5 },
134 			e: { x: bbox.x + bbox.width + this.padding - handlebox.width/4, y: bbox.y + bbox.height/2 },
135 			se: { x: bbox.x + bbox.width + this.padding - handlebox.width/3, y: bbox.y + bbox.height + this.padding - handlebox.height/1.5 },
136 			s: { x: bbox.x + bbox.width/2, y: bbox.y + bbox.height + this.padding - handlebox.height/2 },
137 			sw: { x: bbox.x - this.padding + handlebox.width/3, y: bbox.y + bbox.height + this.padding - handlebox.height/1.5},
138 			w: { x: bbox.x - this.padding + handlebox.width/4, y: bbox.y + bbox.height/2 }
139 		}
140 		// if only one object is selected, the resize box + handles will be rotated according to the object
141 		if (this.data.length == 1) {
142 			// rotate handles manually (-> compute rotated position on the resize box)
143 			var p = { x: bbox.x + bbox.width/2, y: bbox.y + bbox.height/2 };
144 			for (var h in handlePositions) {
145 				var tp = { x: handlePositions[h].x - p.x, y: -(handlePositions[h].y - p.y) };
146 				handlePositions[h].x = p.x + tp.x*Math.cos(angle) - tp.y*Math.sin(angle);
147 				handlePositions[h].y = p.y - (tp.x*Math.sin(angle) + tp.y*Math.cos(angle));
148 			}
149 		}
150 		
151 		this.handles = {
152 			nw: handle.clone().translate(handlePositions.nw.x, handlePositions.nw.y).scale(.5, .5).rotate(45).attr('cursor', 'nw-resize'),
153 			n: handle.clone().translate(handlePositions.n.x, handlePositions.n.y).scale(.5, .5).rotate(90).attr('cursor', 'n-resize'),
154 			ne: handle.clone().translate(handlePositions.ne.x, handlePositions.ne.y).scale(.5, .5).rotate(-45).attr('cursor', 'ne-resize'),
155 			e: handle.clone().translate(handlePositions.e.x, handlePositions.e.y).scale(.5, .5).attr('cursor', 'e-resize'),
156 			se: handle.clone().translate(handlePositions.se.x, handlePositions.se.y).scale(.5, .5).rotate(45).attr('cursor', 'se-resize'),
157 			s: handle.clone().translate(handlePositions.s.x, handlePositions.s.y).scale(.5, .5).rotate(90).attr('cursor', 's-resize'),
158 			sw: handle.clone().translate(handlePositions.sw.x, handlePositions.sw.y).scale(.5, .5).rotate(-45).attr('cursor', 'sw-resize'),
159 			w: handle.clone().translate(handlePositions.w.x, handlePositions.w.y).scale(.5, .5).attr('cursor', 'w-resize')
160 		}
161 		handle.remove();
162 		
163 		// rotate resize box and handles
164 		if (this.data.length == 1) {
165 			var rotDeg = this.data[0].data.rotation,
166 			rotOrigX = bbox.x + bbox.width/2,
167 			rotOrigY = bbox.y + bbox.height/2;
168 			this.resizeBox.rotate(rotDeg, rotOrigX, rotOrigY);
169 			this.handles.w.rotate(rotDeg, true);
170 			this.handles.e.rotate(rotDeg, true);
171 			this.handles.n.rotate(rotDeg, true).rotate(90);
172 			this.handles.s.rotate(rotDeg, true).rotate(90);
173 			this.handles.nw.rotate(rotDeg, true).rotate(45);
174 			this.handles.se.rotate(rotDeg, true).rotate(45);
175 			this.handles.ne.rotate(rotDeg, true).rotate(-45);
176 			this.handles.sw.rotate(rotDeg, true).rotate(-45);
177 		}
178 		
179 		this.scaleInfo = this.canvas.getPaper().text(0, 0, "").attr('font-size', 16).attr('font-weight', 'bold').attr('fill', 'darkblue').hide();
180 		
181 		/** points for resizing */
182 		var scalingPoints = {
183 			// bottom-right
184 			nw: [bbox.x + bbox.width + this.padding / 2, bbox.y + bbox.height + this.padding / 2, bbox.x + bbox.width, bbox.y + bbox.height],
185 			n: [bbox.x + bbox.width + this.padding / 2, bbox.y + bbox.height + this.padding / 2, bbox.x + bbox.width, bbox.y + bbox.height],
186 			// bottom-left
187 			ne: [bbox.x - this.padding / 2, bbox.y + bbox.height + this.padding / 2,  bbox.x, bbox.y + bbox.height],
188 			// top-left
189 			e: [bbox.x - this.padding / 2, bbox.y - this.padding / 2, bbox.x, bbox.y],
190 			se: [bbox.x - this.padding / 2, bbox.y - this.padding / 2, bbox.x, bbox.y],
191 			// top-left
192 			s: [bbox.x - this.padding / 2, bbox.y - this.padding / 2, bbox.x, bbox.y],
193 			// top-right
194 			sw:[bbox.x + bbox.width + this.padding / 2, bbox.y - this.padding / 2, bbox.x + bbox.width, bbox.y],
195 			// bottom-right
196 			w: [bbox.x + bbox.width + this.padding / 2, bbox.y + bbox.height + this.padding / 2, bbox.x + bbox.width, bbox.y + bbox.height]
197 		};
198 		
199 		/** Handles the resize of the resizeBox via the handles */
200 		function handleResize(direction) {
201 			$(that.handles[direction].node).mousedown(function(event) {
202 				// prevent selecting text
203 				event.preventDefault();
204 				event.stopPropagation();
205 				
206 				for (var dir in that.handles) {
207 					if (dir != direction) {
208 						that.handles[dir].hide();
209 					}
210 				}
211 				
212 				// disable connection/annotation mode (icons)
213 				for (var i = 0; i < that.data.length; i++) {
214 					that.data[i].setRotationMode(false);
215 					that.data[i].setConnectionMode(false);
216 					that.data[i].setAnnotationMode(false);
217 					that.data[i].setTextMode(false);
218 				}
219 				that.scaleInfo.show();
220 								
221 				var origX = x = event.pageX,
222 				origY = y = event.pageY,
223 				scaleX = scaleY = 1,
224 				scaleFromX = scaleFromY = 0,
225 				singleObject = (that.data.length == 1),
226 				scaleDirection = direction,
227 				moveEvent = function(event) {
228 					event.preventDefault();
229 					event.stopPropagation();
230 					
231 					that.handles[direction].attr({fill: '#3BB9FF'});
232 					
233 					// compute delta
234 					var delta, normal_vec, deltaX, deltaY;
235 					if (singleObject) {
236 						// if singleObject: object rotation has to be unset before we scale the object (otherwise the object will be rotated around the new center)
237 						that.obj.rotate(0, true);
238 						that.resizeBox.rotate(0, true);
239 						
240 						// no, i'm not going to create ascii art for this
241 						var delta_vec = { x: (event.pageX - origX), y: -(event.pageY - origY) };
242 						var line_vec = { x: -Math.sin(angle), y: Math.cos(angle) }; // "zero"/origin line
243 						var normal_vec = { x: Math.cos(angle), y: Math.sin(angle) }; // normal from the line
244 						
245 						// swap line/normal vector for north/south scale
246 						if (direction == 'n' || direction == 's') {
247 							var tmp = line_vec;
248 							line_vec = normal_vec;
249 							normal_vec = tmp;
250 						}
251 						
252 						// project delta vector onto line vector and compute the distance
253 						var proj_len = (delta_vec.x * line_vec.x + delta_vec.y * line_vec.y);
254 						var proj_vec = { x: proj_len*line_vec.x, y: proj_len*line_vec.y };
255 						var dist_vec = { x: delta_vec.x-proj_vec.x, y: delta_vec.y-proj_vec.y };
256 						delta = Math.sqrt(dist_vec.x*dist_vec.x + dist_vec.y*dist_vec.y);
257 						
258 						// normalize delta vector, compute on which side we're on and multiply with delta
259 						var norm_delta_vec = vonline.vector.normalize(delta_vec);
260 						var side = (norm_delta_vec.x*normal_vec.x + norm_delta_vec.y*normal_vec.y) < 0 ? -1 : 1;
261 						delta *= side;
262 						if (direction.length == 2) {
263 							deltaX = delta;
264 							
265 							// project delta vector onto line vector and compute the distance
266 							var proj_len = (delta_vec.x * normal_vec.x + delta_vec.y * normal_vec.y);
267 							var proj_vec = { x: proj_len*normal_vec.x, y: proj_len*normal_vec.y };
268 							var dist_vec = { x: delta_vec.x-proj_vec.x, y: delta_vec.y-proj_vec.y };
269 							deltaY = Math.sqrt(dist_vec.x*dist_vec.x + dist_vec.y*dist_vec.y);
270 							
271 							// normalize delta vector, compute on which side we're on and multiply with delta
272 							var side = (norm_delta_vec.x*line_vec.x + norm_delta_vec.y*line_vec.y) < 0 ? -1 : 1;
273 							deltaY *= side;
274 							delta = 1;
275 							normal_vec = delta_vec;
276 							if (direction == 'se' || direction == 'nw') {
277 								deltaX = -deltaX;
278 							}
279 						}
280 					}
281 					else {
282 						// if multiple objects are selected, the resize box isn't rotated, so the delta is simply the x or y distance
283 						switch (direction) {
284 							case 'w':
285 							case 'e':
286 								delta = (event.pageX - origX);
287 								normal_vec = { x: 1, y: 0 };
288 								break;
289 							case 'n':
290 							case 's':
291 								delta = -(event.pageY - origY);
292 								normal_vec = { x: 0, y: 1 };
293 								break;
294 							case 'ne':
295 							case 'sw':
296 								deltaX = (event.pageX - origX);
297 								deltaY = -(event.pageY - origY);
298 								normal_vec = { x: deltaX, y: deltaY };
299 								delta = 1;
300 								break;
301 							case 'nw':
302 							case 'se':
303 								deltaX = -(event.pageX - origX);
304 								deltaY = -(event.pageY - origY);
305 								normal_vec = { x: -deltaX, y: deltaY };
306 								delta = 1;
307 								break;
308 						}
309 					}
310 					
311 					// finally: scale
312 					switch (direction) {
313 						case 'w':
314 							delta = -delta;
315 							normal_vec = { x: -normal_vec.x, y: -normal_vec.y };
316 						case 'e':
317 							that.resizeBox.scale(1 + delta / (bbox.width + that.padding), 1, scalingPoints[direction][0], scalingPoints[direction][1]);
318 							that.obj.scale(1 + delta / bbox.width, 1, scalingPoints[direction][2], scalingPoints[direction][3]);						
319 							scaleX = 1 + delta / bbox.width;
320 							break;
321 						case 's':
322 							delta = -delta;
323 							normal_vec = { x: -normal_vec.x, y: -normal_vec.y };
324 						case 'n':
325 							that.resizeBox.scale(1, 1 + delta / (bbox.height + that.padding), scalingPoints[direction][0], scalingPoints[direction][1]);
326 							that.obj.scale(1, 1 + delta / bbox.height, scalingPoints[direction][2], scalingPoints[direction][3]);
327 							scaleY = 1 + delta / bbox.height;
328 							break;
329 						case 'sw':
330 							delta = -delta;
331 							deltaX = -deltaX;
332 							deltaY = -deltaY;
333 							normal_vec = { x: -normal_vec.x, y: -normal_vec.y };
334 						case 'ne':
335 							that.resizeBox.scale(1 + deltaX / (bbox.width + that.padding), 1 + deltaY / (bbox.height + that.padding), scalingPoints[direction][0], scalingPoints[direction][1]);
336 							that.obj.scale(1 + deltaX / bbox.width, 1 + deltaY / bbox.height, scalingPoints[direction][2], scalingPoints[direction][3]);						
337 							scaleX = 1 + deltaX / bbox.width;
338 							scaleY = 1 + deltaY / bbox.height;
339 							break;
340 						case 'se':
341 							delta = -delta;
342 							deltaX = -deltaX;
343 							deltaY = -deltaY;
344 							normal_vec = { x: -normal_vec.x, y: -normal_vec.y };
345 						case 'nw':
346 							that.resizeBox.scale(1 + deltaX / (bbox.width + that.padding), 1 + deltaY / (bbox.height + that.padding), scalingPoints[direction][0], scalingPoints[direction][1]);
347 							that.obj.scale(1 + deltaX / bbox.width, 1 + deltaY / bbox.height, scalingPoints[direction][2], scalingPoints[direction][3]);						
348 							scaleX = 1 + deltaX / bbox.width;
349 							scaleY = 1 + deltaY / bbox.height;
350 							break;
351 					}
352 					
353 					if (singleObject) {
354 						// if singleObject: reset rotation again
355 						that.obj.rotate(that.data[0].data.rotation, bbox.x + bbox.width/2, bbox.y + bbox.height/2);
356 						that.resizeBox.rotate(that.data[0].data.rotation, bbox.x + bbox.width/2, bbox.y + bbox.height/2);
357 					}
358 					// reposition handle
359 					var hbbox = that.handles[direction].getBBox();
360 					that.handles[direction].translate(-hbbox.x-hbbox.width/2, -hbbox.y-hbbox.height/2); // undo current translation
361 					console.log(normal_vec.x * delta, normal_vec.y*delta);
362 					that.handles[direction].translate(handlePositions[direction].x + normal_vec.x*delta, handlePositions[direction].y - normal_vec.y*delta);
363 					
364 					// update scale info and reposition
365 					if (delta != 1 && delta != -1) {
366 						that.scaleInfo.attr({
367 							text: Math.round((scaleX != 1 ? scaleX : scaleY)*100)/100+'x',
368 							x: handlePositions[direction].x + normal_vec.x*(delta+30),
369 							y: handlePositions[direction].y - normal_vec.y*(delta+30)
370 						});
371 					}
372 					else {
373 						that.scaleInfo.hide();
374 					}
375 					
376 					scaleFromX = scalingPoints[direction][2];
377 					scaleFromY = scalingPoints[direction][3];
378 					x = event.pageX;
379 					y = event.pageY;
380 				}
381 				$(window).mousemove(moveEvent);
382 				$(window).one('mouseup', function(event) {
383 					event.preventDefault();
384 					event.stopPropagation();
385 					
386 					$(window).unbind('mousemove', moveEvent);
387 					
388 					var command = new vonline.ScaleCommand(that.data, scaleX, scaleY, scaleFromX, scaleFromY, scaleDirection);
389 					command.execute();
390 					vonline.events.trigger('commandexec', command);
391 					
392 					// reenable connection/annotation mode
393 					for (var i = 0; i < that.data.length; i++) {
394 						$(that.data[i].obj.node).trigger('changed');
395 					}
396 					if (that.data.length == 1) {
397 						that.data[0].setRotationMode(true);
398 						that.data[0].setConnectionMode(true);
399 						that.data[0].setAnnotationMode(true);
400 						that.data[0].setTextMode(true);
401 					}
402 					that.scaleInfo.hide();
403 				});
404 			})
405 			.mouseover(function(event) {
406 				that.handles[direction].attr({fill: '#3BB9FF'});
407 			})
408 			.mouseout(function(event) {
409 				that.handles[direction].attr({fill: 'blue'});
410 			});
411 		}
412 		
413 		handleResize('e');
414 		handleResize('w');
415 		handleResize('n');
416 		handleResize('s');
417 		handleResize('ne');
418 		handleResize('sw');
419 		handleResize('nw');
420 		handleResize('se');
421 	}	
422 }

455 /**
456  * hides the resize box and resize handles
457  * @param {boolean} active
458  */
459 vonline.Selection.prototype.setSelectionMode = function(active) {
460 	if (active) {
461 		if (this.resizeBox) {
462 			this.resizeBox.show();
463 			for (var e in this.handles) {
464 				this.handles[e].show();
465 			}
466 		}
467 	}
468 	else {
469 		if (this.resizeBox) {
470 			this.resizeBox.hide();
471 			for (var e in this.handles) {
472 				this.handles[e].hide();
473 			}
474 		}
475 	}
