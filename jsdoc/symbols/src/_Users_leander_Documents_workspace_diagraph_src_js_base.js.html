1 /**
  2  * Base class for canvas objects
  3  * @namespace
  4  * @constructor
  5  */
  6 vonline.Base = function() {
  7 	// data and object needs to be always in sync
  8 	this.data = vonline.Base.defaultData;
  9 	this.obj = null;
 10 	// overwrite in subclass for connections and annotations
 11 	this.resizeable = true;
 12 	
 13 	this.rotationCircle = null;
 14 	this.rotationHandle = null;
 15 	this.connectionHandle = null;
 16 	this.annotationHandle = null;
 17 	this.inlineTextHandle = null;
 18 	this.connectionPath = null;
 19 }

 21 /** default values and json format specification */
 22 vonline.Base.defaultData = {
 23 	/** unique identifier */
 24 	id: 0,
 25 	/** 'rectangle', 'circle', 'path', 'image', 'annotation', 'connection' */
 26 	type: null,
 27 	/** SVG path or further differentation of type, e.g. 'arrow', 'line' */
 28 	path: null,
 29 	x: 0,
 30 	y: 0,
 31 	width: 50,
 32 	height: 50,
 33 	rotation: 0,
 34 	/** see: http://raphaeljs.com/reference.html#colour */
 35 	color: '90-#ddd-#fff',
 36 	text: null,
 37 	/** [width, color] */
 38 	border: null,
 39 	/** [id, id] */
 40 	connect: []
 41 };

 51 /**
 52  * Displays the element on the canvas
 53  * @param {vonline.Canvas} canvas
 54  */
 55 vonline.Base.prototype.setCanvas = function(canvas) {
 56 	if (!canvas && this.obj) {
 57 		// remove
 58 		this.obj.remove();
 59 		if (this.text) {
 60 			this.text.remove();
 61 		}
 62 		return;
 63 	}
 64 	this.canvas = canvas;
 65 	this.obj = this.createObject(canvas).initZoom();
 66 	this.setPosition(this.data.x, this.data.y);
 67 	this.setRotation(this.data.rotation);
 68 	this.setColor(this.data.color);
 69 	this.setBorder();
 70 	this.createText(canvas);
 71 	
 72 	this.setClickEventMode(true);
 73 	this.setDragEventMode(this);
 74 	
 75 	this.obj.node.id = this.createObjectId();
 76 }

 85 /**
 86  * Set the position of the component
 87  * Note that this function sets the absolute position by the top-left corner of the bounding box
 88  * @param {integer} x
 89  * @param {integer} y
 90  */
 91 vonline.Base.prototype.setPosition = function(x, y) {
 92 	var bbox = this.obj.getBBox(),
 93 		currentX = bbox.x,
 94 		currentY = bbox.y;
 95 	this.obj.translate(x - currentX, y - currentY);
 96 	this.data.x = x;
 97 	this.data.y = y;
 98 }

146 /**
147  * creates a text within the object
148  */
149 vonline.Base.prototype.createText = function() {
150 	var that = this;
151 	
152 	if (this.data.text) {
153 		this.text = this.canvas.getPaper().text(this.data.x, this.data.y, this.data.text).initZoom();
154 		this.text.setAttr('font-size', 10);
155 		
156 		// redirect events to object
157 		$(this.text.node).click(function(event) {
158 			$(that.obj.node).trigger(event);
159 		});
160 		$(this.text.node).dblclick(function(event) {
161 			$(that.obj.node).trigger(event);
162 		});
163 		$(this.text.node).mousedown(function(event) {
164 			$(that.obj.node).trigger(event);
165 		});
166 		$(this.text.node).mousemove(function(event) {
167 			// fix for connection-path
168 			event.target = that.obj.node;
169 			$(that.obj.node).trigger(event);
170 		});
171 		
172 		// reposition on change
173 		this.adjustText();
174 		$(this.obj.node).bind('changed', function() {
175 			that.adjustText();
176 		});
177 		$(vonline.events).bind('zoom', function() {
178 			that.adjustText();
179 		});
180 		
181 		$(this.obj.node).trigger('textchanged');
182 	}
183 }

206 /**
207  * Returns the JSON representation of the object
208  * @return {json} this.data 
209  */
210 vonline.Base.prototype.toJSON = function() {
211 	// clean up (remove default values)
212 	var json = {};
213 	for (var property in this.data) {
214 		if (this.data[property] != vonline.Base.defaultData[property]) {
215 			json[property] = this.data[property];
216 		}
217 	}
218 	return json;
219 }

225 vonline.Base.prototype.setSelection = function(active, selected) {
226 	if (active) {
227 		this.obj.attr('stroke', 'red');
228 		this.selected = true;

230 		
231 		this.updateSelection(selected == 1);
232 		var that = this;
233 	}
234 	else {
235 		this.obj.attr('stroke', 'black');
236 		this.selected = false;
237 		$(this.obj.node).trigger('unselect');
238 		
239 		this.updateSelection();
240 	}
241 	if (!this.selectionHandler) {
242 		this.selectionHandler = vonline.events.bind('selection_changed', function(event, number) {
243 			that.updateSelection(number == 1);
244 		});
245 	}

266 /**
267  * @param {vonline.Base / array} objects
268  * @param {function} ondrag (optional) is executed on drag (e.g. for updating interface)
269  */
270 vonline.Base.prototype.setDragEventMode = function(objects, ondrag) {
271 	if (!$.isArray(objects)) {
272 		objects = [objects];
273 	}
274 	
275 	var that = this;
276 	this.obj.attr('cursor', 'pointer');
277 	
278 	if (this.dragEventHandler) {
279 		$(this.obj.node).unbind('mousedown', this.dragEventHandler);
280 	}
281 	/**
282 	 * catches the mouse events
283 	 */
284 	this.dragEventHandler = function(event) {
285 		that.obj.attr('cursor', 'move');
286 		
287 		$.each(objects, function(i, object) {
288 			object.setRotationMode(false);
289 			object.setConnectionMode(false);
290 			object.setAnnotationMode(false);
291 			object.setTextMode(false);
292 		});
293 		
294 		// prevent selecting text
295 		event.preventDefault();
296 		event.stopPropagation();
297 		
298 		var origX = x = event.pageX,
299 			origY = y = event.pageY;
300 		var moveEvent = function(event) {
301 			// distinguish between click and drag-event, see setClickHandler 
302 			that.wasDragging = true;
303 			that.setClickEventMode(false);
304 			
305 			event.preventDefault();
306 			event.stopPropagation();
307 			
308 			var deltaX = event.pageX - x,
309 				deltaY = event.pageY - y;
310 			if (event.altKey) {
311 				$.each(objects, function(i, object) {
312 					var bbox = object.obj.getBBox();
313 					// adjust object
314 					object.obj.translate((Raphael.snapTo(vonline.GRIDSIZE, bbox.x) - bbox.x), (Raphael.snapTo(vonline.GRIDSIZE, bbox.y) - bbox.y));
315 				});
316 				event.pageX = Raphael.snapTo(vonline.GRIDSIZE, event.pageX);
317 				deltaX = Raphael.snapTo(vonline.GRIDSIZE, event.pageX - x);
318 				event.pageY = Raphael.snapTo(vonline.GRIDSIZE, event.pageY);
319 				deltaY = Raphael.snapTo(vonline.GRIDSIZE, event.pageY - y);
320 			}
321 			$.each(objects, function(i, object) {
322 				object.setTranslation(deltaX, deltaY);
323 			});
324 			x = event.pageX;
325 			y = event.pageY;
326 			
327 			if (ondrag) {
328 				ondrag();
329 			}
330 		} 
331 		$(window).mousemove(moveEvent);
332 		$(window).one('mouseup', function(event) {
333 			event.preventDefault();
334 			event.stopPropagation();
335 			
336 			$(window).unbind('mousemove', moveEvent);
337 			if (that.wasDragging) {
338 				var translateX = (x - origX),
339 					translateY = (y - origY);
340 				vonline.events.trigger('commandexec', new vonline.TranslateCommand(objects, translateX, translateY));
341 				that.wasDragging = false;
342 				window.setTimeout(function() {
343 					that.setClickEventMode(true);
344 				}, 0);
345 			}
346 			that.obj.attr('cursor', 'pointer');
347 			
348 			$.each(objects, function(i, object) {
349 				if (object.selected) {
350 					object.updateSelection(objects.length == 1);
351 					$(object.obj.node).trigger('changed');
352 				}
353 			});
354 		});
355 	};
356 	$(this.obj.node).mousedown(this.dragEventHandler);
357 }

359 /**
360  * @param {boolean} active
361  */
362 vonline.Base.prototype.setClickEventMode = function(active) {
363 	var that = this;
364 	if (active) {
365 		this.clickEventHandler = function(event) {
366 			event.preventDefault();
367 			event.stopPropagation();
368 			
369 			if (event.shiftKey) {
370 				vonline.document.canvas.selection.toggle(that);
371 			}
372 			else {
373 				vonline.document.canvas.selection.clear();
374 				vonline.document.canvas.selection.add(that);
375 			}
376 		}
377 		$(this.obj.node).click(this.clickEventHandler);
378 	}
379 	else {
380 		$(this.obj.node).unbind('click', this.clickEventHandler);
381 	}
382 }

384 /**
385  * @param {boolean} active
386  */
387 vonline.Base.prototype.setRotationMode = function(active) {
388 	var bbox = this.obj.getBBox(),
389 	radius = Math.sqrt(Math.pow(bbox.width / 2, 2) + Math.pow(bbox.height / 2, 2)) + 20,
390 	centerX = bbox.x + bbox.width / 2,
391 	centerY = bbox.y + bbox.height / 2,
392 	that = this;
393 	
394 	
395 	if (this.rotationCircle || this.rotationHandle) {
396 		$(this.obj.node).unbind('changed', this.updateRotationHandles);
397 		if (this.rotationCircle) {
398 			this.rotationCircle.remove();
399 			this.rotationCircle = null;
400 		}
401 		if (this.rotationHandle) {
402 			this.rotationHandle.remove();
403 			this.rotationHandle = null;
404 		}
405 	}
406 	if (active) {
407 		/**
408 		 * updates rotation-circle and rotation-handle on object changes
409 		 */
410 		this.updateRotationHandles = function() {
411 			var newBBox = that.obj.getBBox(),
412 			newRadius = Math.sqrt(Math.pow(newBBox.width / 2, 2) + Math.pow(newBBox.height / 2, 2)) + 20,
413 			newCenterX = newBBox.x + newBBox.width / 2,
414 			newCenterY = newBBox.y + newBBox.height / 2;
415 			if (that.rotationCircle) {
416 				if (radius != newRadius) {
417 					that.rotationCircle.scale(newRadius/radius, newRadius/radius);
418 					that.rotationCircle.resetScale();
419 					that.rotationCircle.translate(newCenterX - centerX, newCenterY - centerY);
420 					that.rotationHandle.rotate(0, centerX, centerY);
421 					that.rotationHandle.translate(newCenterX - centerX, newCenterY - newRadius - (centerY - radius));
422 					that.rotationHandle.rotate(that.data.rotation, newCenterX, newCenterY);
423 					radius = newRadius;
424 					centerX = newCenterX;
425 					centerY = newCenterY;
426 				}
427 				else if (centerX != newCenterX || centerY != newCenterY) {
428 					that.rotationCircle.translate(newCenterX - centerX, newCenterY - centerY);
429 					that.rotationHandle.translate(newCenterX - centerX, newCenterY - newRadius - (centerY - radius));
430 					centerX = newCenterX;
431 					centerY = newCenterY;
432 				}
433 			}
434 		}
435 		$(this.obj.node).bind('changed', this.updateRotationHandles);
436 		
437 		this.rotationCircle = this.canvas.getPaper().circle(centerX, centerY, radius).attr('stroke', 'orange').attr('stroke-width', '1');
438 		this.rotationHandle = this.canvas.getPaper().circle(centerX, centerY - radius, 5.5).attr('stroke', 'none').attr('fill', 'orange').attr('cursor', 'pointer').rotate(this.data.rotation, centerX, centerY);
439 		this.rotationInfo = this.canvas.getPaper().text(0, 0, "").attr('font-size', 16).attr('font-weight', 'bold').attr('fill', 'orangered').hide();
440 		$(this.rotationHandle.node).mousedown(function(event) {
441 			event.stopPropagation();
442 			
443 			// disable/hide everything, but the rotation controls
444 			that.setConnectionMode(false);
445 			that.setAnnotationMode(false);
446 			that.setTextMode(false);
447 			that.canvas.selection.setSelectionMode(false);
448 			that.rotationInfo.show();
449 			that.rotationCircle.attr('stroke-width', '2');
450 			
451 			var deg = that.data.rotation,
452 			origDeg = that.data.rotation;
453 			
454 			function rotationEvent(event) {
455 				event = that.canvas.normalizeEvent(event);
456 				deg = Raphael.angle(centerX, centerY, event.offsetX, event.offsetY);
457 				deg -= 90;
458 				deg = deg < 0 ? 360 + deg : deg;
459 				if (!event.altKey) {
460 					deg = Raphael.snapTo(45, deg, 20);
461 				}
462 				if (deg == 360) {
463 					deg = 0;
464 				}
465 				that.rotationHandle.rotate(deg, centerX, centerY);
466 				that.obj.rotate(deg, true);
467 				that.rotationHandle.attr({fill: 'yellow'});
468 				
469 				var angle = -deg * Math.PI/180; 
470 				var infoPos = { x: -(radius+25)*Math.sin(angle), y: -(radius+25)*Math.cos(angle) };
471 				that.rotationInfo.attr({text: Math.round(deg)+'°', x: centerX+infoPos.x, y: centerY+infoPos.y});
472 			}
473 			
474 			$(window).mousemove(rotationEvent);
475 			$(window).one('mouseup', function() {
476 				$(window).unbind('mousemove', rotationEvent);
477 				if (deg != origDeg) {
478 					var command = new vonline.RotateCommand(that, deg);
479 					command.execute();
480 					vonline.events.trigger('commandexec', command);
481 				}
482 				
483 				// enable/show everything again
484 				that.setConnectionMode(true);
485 				that.setAnnotationMode(true);
486 				that.setTextMode(true);
487 				that.canvas.selection.setSelectionMode(true);
488 				
489 				that.rotationHandle.attr({fill: 'orange'});
490 				that.rotationInfo.hide();
491 				that.rotationCircle.attr('stroke-width', '1');
492 			});
493 		})
494 		.mouseover(function(event) {
495 			that.rotationHandle.attr({fill: 'yellow'});
496 		})
497 		.mouseout(function(event) {
498 			that.rotationHandle.attr({fill: 'orange'});
499 		});
500 	}
501 }

503 /**
504  * show connection handles
505  */
506 vonline.Base.prototype.setConnectionMode = function(active) {
507 	var that = this;
508 	
509 	if (this.connectionHandle) {
510 		$(this.obj.node).unbind('changed', this.updateConnectionHandles);
511 		this.connectionHandle.remove();
512 		this.connectionHandle = null;
513 	}
514 	if (active) {
515 		var bbox = this.obj.getBBox();
516 		this.updateConnectionHandles = function() {
517 			if (that.connectionHandle) {
518 				var newBBox = that.obj.getBBox();
519 				that.connectionHandle.translate(newBBox.x - bbox.x + newBBox.width - bbox.width, newBBox.y - bbox.y + newBBox.height - bbox.height);
520 				that.connectionHandle.rotate(that.data.rotation, newBBox.x + newBBox.width/2, newBBox.y + newBBox.height/2);
521 				bbox = newBBox;
522 			}
523 		}
524 		$(this.obj.node).bind('changed', this.updateConnectionHandles);
525 		
526 		// create connection handle
527 		this.connectionHandle = this.canvas.getPaper().image('images/canvas/connect.png', 0, 0, 22, 22).attr('cursor', 'pointer');
528 		var handleBBox = this.connectionHandle.getBBox();
529 		this.connectionHandle.translate(bbox.x + bbox.width + 20, bbox.y + 38);
530 		this.connectionHandle.rotate(this.data.rotation, bbox.x + bbox.width/2, bbox.y + bbox.height/2);
531 		
532 		// handles events
533 		if(this.connectionPath == null) {
534 			this.connectionPath = this.canvas.getPaper().path('M0,0').toBack();
535 		}
536 		this.connectionPath.attr('path', 'M0,0').hide();
537 		
538 		var targetObj = null, targetStrokeColor = null, targetStrokeWidth = null;
539 		var restoreTargetObj = function() {
540 			if(targetObj) {
541 				targetObj.obj.attr('stroke', (targetStrokeColor ? targetStrokeColor : '#000'));
542 				targetObj.obj.attr('stroke-width', (targetStrokeWidth ? targetStrokeWidth : '1'));
543 				targetObj = null, targetStrokeColor = null, targetStrokeWidth = null;
544 			}
545 		}
546 		var connectionMoveEvent = function(event) {
547 			event.preventDefault();
548 			event.stopPropagation();
549 			event = that.canvas.normalizeEvent(event);
550 			
551 			that.connectionPath.show();
552 			
553 			if ($(event.target).is('tspan')) {
554 				// fix for wrong connection when mousepointer over test
555 				return;
556 			}
557 			
558 			// highlight target object
559 			if (event.target.id && event.target.id.substr(0, 7) == 'object_' && that.obj.node.id != event.target.id) {
560 				var newTargetObj = that.canvas.getObjectById(event.target.id.replace(/object_/, ''));
561 				
562 				if (newTargetObj != targetObj) {
563 					// if an old target object exists, restore old values
564 					restoreTargetObj();
565 					
566 					targetObj = newTargetObj;
567 					targetStrokeColor = targetObj.obj.attr('stroke');
568 					targetStrokeWidth = targetObj.obj.attr('stroke-width');
569 				}
570 				
571 				targetObj.obj.attr('stroke', 'red').attr('stroke-width', '2');
572 				
573 				// update path
574 				var targetBBox = targetObj.obj.getBBox();
575 				that.connectionPath.attr('path', vonline.Connection.computePath(bbox.x+bbox.width/2, bbox.y+bbox.height/2, targetBBox.x+targetBBox.width/2, targetBBox.y+targetBBox.height/2));
576 			}
577 			else {
578 				restoreTargetObj();
579 				
580 				// update path
581 				that.connectionPath.attr('path', vonline.Connection.computePath(bbox.x+bbox.width/2, bbox.y+bbox.height/2, event.offsetX, event.offsetY));
582 			}
583 		}
584 		$(this.connectionHandle.node).mousedown(function(event) {
585 			event.stopPropagation();
586 			$(window).bind('mousemove', connectionMoveEvent);
587 		});
588 		$(this.connectionHandle.node).click(function(event) {
589 			event.stopPropagation();
590 			
591 			that.canvas.selection.setConnectionMode(true);
592 			that.canvas.container.one('click', function(event) {
593 				$(window).unbind('mousemove', connectionMoveEvent);
594 				that.connectionPath.hide();
595 				restoreTargetObj(); // if a target object exists, restore old values
596 				
597 				that.canvas.selection.setConnectionMode(false);
598 				// abort if user clicks on canvas or the same object
599 				if (event.target.id && event.target.id.substr(0, 7) == 'object_' && that.obj.node.id != event.target.id) {
600 					// see vonline.Document
601 					vonline.events.trigger('drop', {type:'connection', connect: [that.obj.node.id.replace(/object_/, ''), event.target.id.replace(/object_/, '')]});
602 				}
603 			});
604 		});
605 	}
606 }

608 /**
609  * show annotation handles
610  */
611 vonline.Base.prototype.setAnnotationMode = function(active) {
612 	var that = this
613 	
614 	if(this.annotationHandle) {
615 		$(this.obj.node).unbind('changed', this.updateAnnotationHandle);
616 		this.annotationHandle.remove();
617 		this.annotationHandle = null;
618 	}
619 	if (active) {
620 		var bbox = this.obj.getBBox();
621 		this.updateAnnotationHandle = function() {
622 			if (that.annotationHandle) {
623 				var newBBox = that.obj.getBBox();

627 			}
628 		}
629 		$(this.obj.node).bind('changed', this.updateAnnotationHandle);
630 		
631 		// create connection handle
632 		this.annotationHandle = this.canvas.getPaper().image('images/canvas/annotate.png', 0, 0, 22, 22).attr('cursor', 'pointer');

636 		
637 		// handles events

640 		});
641 		$(this.annotationHandle.node).click(function(event) {
642 			event.stopPropagation();

647 					vonline.events.trigger('drop', {type:'annotation', text: annotation, connect: [that.data.id], x: that.data.width + 20, y: 0});
648 				}
649 			}});
650 		});
651 	}

654 /**
655  * enables/disables the object text show/hide event
656  */
657 vonline.Base.prototype.setTextMode = function(active) {
658 	var that = this
659 	
660 	if (this.inlineTextHandle) {
661 		$(this.obj.node).unbind('changed', this.updateInlineTextHandle);
662 		this.inlineTextHandle.remove();

664 	}
665 	if (active) {

667 		this.updateInlineTextHandle = function() {
668 			if (that.inlineTextHandle) {
669 				var newBBox = that.obj.getBBox();
670 				that.inlineTextHandle.translate(newBBox.x - bbox.x + newBBox.width - bbox.width, newBBox.y - bbox.y + newBBox.height - bbox.height);
671 				that.inlineTextHandle.rotate(that.data.rotation, newBBox.x + newBBox.width/2, newBBox.y + newBBox.height/2);
672 				bbox = newBBox;
673 			}
674 		}
675 		$(this.obj.node).bind('changed', this.updateInlineTextHandle);
676 		
677 		// create connection handle
678 		this.inlineTextHandle = this.canvas.getPaper().image('images/canvas/inline_text.png', 0, 0, 22, 22).attr('cursor', 'pointer');
679 		var handleBBox = this.inlineTextHandle.getBBox();
680 		this.inlineTextHandle.translate(bbox.x + bbox.width + 20, bbox.y + 14);

686 		});
687 		$(this.inlineTextHandle.node).click(function(event) {

699 /**
700  * Checks if bounding contains the current object
701  * @param {object} bounding Specified by (x, y, width, height)
702  */
703 vonline.Base.prototype.fitsIn = function(bounding) {
704 	var bbox = this.obj.getBBox();
705 	return (bbox.x >= bounding.x && bbox.y >= bounding.y
706 			&& bbox.x + bbox.width <= bounding.x + bounding.width
707 			&& bbox.y + bbox.height <= bounding.y + bounding.height);
708 }

710 /**
711  * tests if an object with given id is connected to the current object
712  * @param {vonline.Base} obj
713  */
714 vonline.Base.prototype.isConnectedTo = function(obj) {
715 	for (var i = 0, len = this.data.connect.length; i < len; i++) {
716 		if (this.data.connect[i] == obj.getId()) {
717 			return true;
718 		}
719 	}
720 	return false;
