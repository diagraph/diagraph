1 /**
  2  * @param {vonline.Document} document
  3  * @param {vonline.Sidebar} sidebar
  4  * @param {vonline.Transport} transport
  5  */
  6 vonline.SnapshotHistory = function(document, sidebar, transport) {
  7 	this.document = document;
  8 	this.sidebar = sidebar;
  9 	this.transport = transport;
 10 	this.container = null;
 11 	this.snapshotList = null;
 12 	this.snapshotListItems = null;
 13 	this.selectedSnapshot = -1;
 14 	
 15 	//
 16 	var that = this;
 17 	this.loadButton = new vonline.MenuItem('load snapshot', 'images/menu/revert', function() {
 18 		that.loadSnapshot();
 19 	});
 20 	this.loadButton.getHTML().css('margin-left', '90px');
 21 	
 22 	//
 23 	this.history = $('<div/>').addClass('history');
 24 	
 25 	//
 26 	vonline.events.bind('snapshotsaved', function() {
 27 		that.update();
 28 	});
 29 }
 30 
 31 vonline.SnapshotHistory.prototype.open = function() {
 32 	if(this.sidebar.isExtraViewVisible()) return;
 33 	
 34 	// init menu, enable everything
 35 	this.sidebar.getTopMenu().addItem(this.loadButton);
 36 	this.container = this.sidebar.setExtraView(true);
 37 	this.container.append(this.history);
 38 	this.history.show();
 39 	this.update();
 40 	this.loadButton.disable();
 41 }
 42 
 43 vonline.SnapshotHistory.prototype.update = function() {
 44 	if(this.container == null) return;
 45 
 46 	// clear history and get/add new snapshots
 47 	var that = this;
 48 	this.history.empty();
 49 	this.transport.getSnapshots(function(json) {
 50 		var num = 1;
 51 		that.snapshotList = $('<ul/>').addClass('snapshotList');
 52 		that.snapshotListItems = [];
 53 		$.each(json, function(i, e) {
 54 			that.snapshotListItems.push(
 55 					$('<li/>')
 56 					.addClass('snapshotListItem')
 57 					.attr('id', 'snapshot'+e.id)
 58 					.attr('title', e.id)
 59 					.append('#'+num+' '+e.creation_date)
 60 					.click(function(event) {
 61 						that.selectSnapshot(event.currentTarget.id);
 62 					})
 63 					.append($('<img/>').attr('src', 'images/canvas/edit_delete.png')
 64 							.click(function() {that.deleteSnapshot(e.id);}))
 65 				);
 66 				
 67 				that.snapshotList.prepend(that.snapshotListItems[that.snapshotListItems.length-1]);
 68 				num++;
 69 		});
 70 		that.history.append(that.snapshotList);
 71 	});
 72 }
 73 
 74 vonline.SnapshotHistory.prototype.close = function() {
 75 	this.sidebar.getTopMenu().removeItem(this.loadButton);
 76 	this.sidebar.setExtraView(false);
 77 	this.container = null;
 78 	this.history.hide();
 79 	this.history.detach();
 80 	this.snapshotList = null;
 81 	this.snapshotListItems = null;
 82 	this.selectedSnapshot = -1;
 83 }
 84 
 85 /**
 86  * @return {boolean} if view was closed
 87  */
 88 vonline.SnapshotHistory.prototype.toggle = function() {
 89 	var ret = false;
 90 	if (this.container) {
 91 		this.close();
 92 		ret = true;
 93 	}
 94 	else if(!this.sidebar.isExtraViewVisible()) {
 95 		this.open();
 96 		ret = false;
 97 	}
 98 	$(window).trigger('resize');
 99 	return ret;
100 }
101 
102 vonline.SnapshotHistory.prototype.loadSnapshot = function() {
103 	if (this.selectedSnapshot != -1) {
104 		this.document.loadSnapshot(this.selectedSnapshot);
105 	}
106 	this.toggle();
107 }
108 
109 vonline.SnapshotHistory.prototype.selectSnapshot = function(id) {
110 	if (!this.snapshotListItems) return;
111 	
112 	for (var i = 0; i < this.snapshotListItems.length; i++) {
113 		this.snapshotListItems[i].removeClass('selected');
114 		if(this.snapshotListItems[i].attr('id') == id) {
115 			this.snapshotListItems[i].addClass('selected');
116 			this.selectedSnapshot = parseInt(this.snapshotListItems[i].attr('title'));
117 		}
118 	}
119 	this.loadButton.enable();
120 }
121 
122 vonline.SnapshotHistory.prototype.deleteSnapshot = function(id) {
123 	this.transport.deleteSnapshot(id);
124 }
