1 /**
  2  * @namespace
  3  * @augments vonline.Base
  4  */
  5 vonline.Annotation = function(data) {
  6 	// inherit default values
  7 	this.data = $.extend({}, this.data, data);
  8 	this.resizeable = false;
  9 }
 10 
 11 vonline.Annotation.prototype = new vonline.Base();
 12 
 13 vonline.Annotation.prototype.createObjectId = function() {
 14 	return 'annotation_'+this.data.id;
 15 }
 16 
 17 /**
 18  * Creates the object on the given canvas
 19  * @param {vonline.Canvas} canvas
 20  */
 21 vonline.Annotation.prototype.createObject = function(canvas) {
 22 	var that = this;
 23 	
 24 	this.sourceObject = canvas.getObjectById(this.data.connect[0]);
 25 	var bbox = this.sourceObject.obj.getBBox();
 26 	
 27 	// event handler
 28 	$(this.sourceObject.obj.node).bind('changed', function() {
 29 		that.updateLayout();
 30 	});
 31 	var obj = canvas.getPaper().rect(bbox.x + this.data.x, bbox.y + this.data.y, 100, 15).attr('stroke-dasharray', '--').attr('fill', 'rgba(255,255,255,0.7)');
 32 	
 33 	$(obj.node).bind('textchanged', function() {
 34 		that.updateLayout();
 35 	});
 36 	
 37 	$(obj.node).dblclick(function(event) {
 38 		event.stopPropagation();
 39 		
 40 		new vonline.InputDialog({text: 'Enter a new inline text:', confirm: function(inlinetext) {
 41 			var command = new vonline.TextChangeCommand(that, inlinetext);
 42 			command.execute();
 43 			vonline.events.trigger('commandexec', command);
 44 		}});
 45 	});
 46 	
 47 	return obj;
 48 }
 49 
 50 vonline.Annotation.prototype.updateLayout = function() {
 51 	var textwidth = Math.ceil(this.text.getBBox().width);
 52 	this.obj.attr('width', textwidth + 10);
 53 	this.setPosition(this.data.x, this.data.y);
 54 }
 55 
 56 vonline.Annotation.prototype.setPosition = function(x, y) {
 57 	var bbox = this.sourceObject.obj.getBBox(),
 58 	currentX = this.obj.attr('x'),
 59 	currentY = this.obj.attr('y');
 60 	
 61 	this.obj.translate(bbox.x + x - currentX, bbox.y + y - currentY);
 62 	this.data.x = x;
 63 	this.data.y = y;
 64 	$(this.obj.node).trigger('changed');
 65 }
