1 /**
  2  * @namepsace 
  3  */
  4 vonline.Notification = function() {
  5 	this.container = $('<div/>').addClass('notification').appendTo('#canvas');
  6 }
  7 
  8 /**
  9  * @param {string} message
 10  * @param {object} options
 11  */
 12 vonline.Notification.prototype.add = function(message, options) {
 13 	options = $.extend({
 14 		timeout: 5,
 15 		callback: null,
 16 		init: null
 17 	}, options);
 18 	var e = $('<div/>').text(message).appendTo(this.container),
 19 	that = this;
 20 	
 21 	if (options.init) {
 22 		options.init(e);
 23 	}
 24 	if (options.timeout > 0) {
 25 		var timeouthandler = this.setTimeout(e, options.timeout);
 26 		e.mouseover(function() {
 27 			window.clearTimeout(timeouthandler);
 28 		});
 29 		e.mouseout(function() {
 30 			timeouthandler = that.setTimeout(e, 2);
 31 		});
 32 	}
 33 	if (options.callback) {
 34 		e.click(function() {
 35 			options.callback();
 36 			e.fadeOut(function(){$(this).detach();});
 37 		});
 38 	}
 39 	else if (options.timeout > 0) {
 40 		e.click(function() {
 41 			e.fadeOut(function(){$(this).detach();});
 42 		});
 43 	}
 44 }
 45 
 46 vonline.Notification.prototype.setTimeout = function(e, timeout) {
 47 	return window.setTimeout(function() {
 48 		e.fadeOut(function(){$(this).detach();});
 49 	}, timeout*1000);
 50 }
